import sqlite3, tempfile, os

# Atomicity test: simulate mid-migration failure
with tempfile.NamedTemporaryFile(suffix='.db', delete=False) as f:
    test_path = f.name

# Create a db at version 1 (tables exist, no extra columns)
conn = sqlite3.connect(test_path)
conn.executescript('''
CREATE TABLE algo_snapshots (id INTEGER PRIMARY KEY, refreshed_at TEXT NOT NULL, source_names TEXT NOT NULL, algorithm_model TEXT NOT NULL, signal_weights TEXT, longform_insights TEXT, confidence TEXT DEFAULT 'medium');
CREATE TABLE competitor_channels (channel_id TEXT PRIMARY KEY, channel_name TEXT NOT NULL, channel_url TEXT, subscriber_count INTEGER, niche_category TEXT, track_active INTEGER DEFAULT 1, added_at TEXT NOT NULL);
CREATE TABLE competitor_videos (video_id TEXT PRIMARY KEY, channel_id TEXT, title TEXT NOT NULL, published_at TEXT NOT NULL, views INTEGER, likes INTEGER, duration_seconds INTEGER, description TEXT, is_outlier INTEGER DEFAULT 0, outlier_reason TEXT, fetched_at TEXT NOT NULL);
CREATE TABLE niche_snapshots (id INTEGER PRIMARY KEY, refreshed_at TEXT NOT NULL, format_patterns TEXT, hook_patterns TEXT, trending_topics TEXT);
CREATE TABLE kb_meta (id INTEGER PRIMARY KEY, last_refresh TEXT, last_export TEXT, version INTEGER DEFAULT 1);
''')
conn.execute('PRAGMA user_version = 1')
conn.commit()
conn.close()

# Simulate failure: add topic_cluster then fail before outlier_ratio
conn = sqlite3.connect(test_path)
try:
    with conn:
        conn.execute('ALTER TABLE competitor_videos ADD COLUMN topic_cluster TEXT')
        raise RuntimeError('simulated mid-migration failure')
except RuntimeError:
    pass

# Verify rollback: topic_cluster should NOT exist
cols = [r[1] for r in conn.execute('PRAGMA table_info(competitor_videos)').fetchall()]
assert 'topic_cluster' not in cols, f'Rollback failed — topic_cluster exists: {cols}'

# Verify version unchanged
v = conn.execute('PRAGMA user_version').fetchone()[0]
assert v == 1, f'Version should still be 1, got {v}'
conn.close()
os.unlink(test_path)

print('Atomicity rollback test passed')
