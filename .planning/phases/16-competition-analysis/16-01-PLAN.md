   def _ensure_classification_columns(self):
       """Add classification columns if they don't exist (safe for existing DBs)."""
       cursor = self._conn.cursor()

       # Check if columns exist
       cursor.execute("PRAGMA table_info(competitor_videos)")
       existing = {row[1] for row in cursor.fetchall()}

       # Add missing columns
       if 'format' not in existing:
           cursor.execute("ALTER TABLE competitor_videos ADD COLUMN format TEXT")
       if 'angles' not in existing:
           cursor.execute("ALTER TABLE competitor_videos ADD COLUMN angles TEXT")
       if 'quality_tier' not in existing:
           cursor.execute("ALTER TABLE competitor_videos ADD COLUMN quality_tier TEXT")
       if 'classified_at' not in existing:
           cursor.execute("ALTER TABLE competitor_videos ADD COLUMN classified_at DATE")

       self._conn.commit()
