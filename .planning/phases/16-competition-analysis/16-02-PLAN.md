```python
def filter_quality_competition(
    videos: List[Dict],
    min_views: int = 1000,
    min_percentile: int = 25
) -> List[Dict]:
    """
    Filter to quality competition only.

    Quality signals (available without API quota):
    1. View count > threshold (1K minimum for educational content)
    2. View count > Nth percentile of sample (relative quality)
    3. Channel appears multiple times (established creator, not spam)

    Args:
        videos: List of video dicts with 'view_count' and 'channel_name' keys
        min_views: Absolute minimum view count (default 1000)
        min_percentile: Percentile threshold 0-100 (default 25th)

    Returns:
        Filtered list with quality_signals dict added to each video
    """
```

Implementation per RESEARCH.md Pattern 3:
- Calculate view count percentile threshold from sample
- Effective threshold = max(min_views, percentile_threshold)
- Count channel occurrences for established creator detection
- Add quality_signals dict to each passing video:

  - 'channel_video_count': int
  - 'is_established_creator': bool (2+ videos from channel)
- Assign quality_tier: 'high' (>75th percentile), 'medium' (25-75th), 'low' (below)

```python
def calculate_differentiation_score(
    videos: List[Dict],
    channel_angles: List[str] = None
) -> Dict[str, Any]:
    """
    Calculate differentiation opportunities based on angle frequency.

    Differentiation = angles that are UNDERREPRESENTED in competition.
    Higher score = bigger gap = better opportunity for this channel.

    Args:
        videos: List of video dicts with 'angles' key (list of angle strings)
        channel_angles: This channel's preferred angles (default: ['legal', 'historical'])

    Returns:
        {
            'angle_distribution': {'political': 45.2, 'legal': 12.1, ...},  # Percentages
            'gap_scores': {'legal': 0.73, 'economic': 0.85, ...},  # 1.0 - frequency
            'recommended_angle': 'economic',  # Best gap for channel
            'differentiation_score': 0.85,  # Score for recommended angle
            'total_videos_analyzed': int
        }
    """
```

Implementation per RESEARCH.md Pattern 4:
- Default channel_angles to ['legal', 'historical'] (channel DNA)
- Count angle frequency across all videos
- Calculate gap_scores as inverse of frequency (1.0 - frequency)
- Include all 5 angle categories even if 0 occurrences
- Find best angle from channel_angles based on gap score
- Return 1.0 differentiation if no competition (maximum opportunity)

```python
def main():
    """CLI entry point for competition analysis."""
    import argparse

    parser = argparse.ArgumentParser(
        description='Analyze competition for a keyword'
    )
    parser.add_argument('keyword', help='Keyword to analyze')
    parser.add_argument('--sample-size', type=int, default=100,
                        help='Max videos to sample (default 100)')
    parser.add_argument('--quality-only', action='store_true',
                        help='Show only quality competition')
    parser.add_argument('--json', action='store_true',
                        help='Output as JSON')
    parser.add_argument('-v', '--verbose', action='store_true',
                        help='Show detailed output')

    args = parser.parse_args()

    analyzer = CompetitionAnalyzer(sample_size=args.sample_size)
    result = analyzer.analyze_competition(args.keyword)

    if args.json:
        import json
        print(json.dumps(result, indent=2, default=str))
        return

    if 'error' in result:
        print(f"Error: {result['error']}")
        if 'details' in result:
            print(f"Details: {result['details']}")
        return

    # Pretty print results
    print(f"\nCompetition Analysis: {args.keyword}")
    print("=" * 50)
    print(f"Total Videos: {result['video_count']}")
    print(f"Unique Channels: {result['channel_count']}")
    print(f"Quality Videos: {result['quality_video_count']}")
    print()

    print("Format Breakdown:")
    total = sum(result['format_breakdown'].values()) or 1
    for fmt, count in sorted(result['format_breakdown'].items()):
        pct = count / total * 100
        print(f"  {fmt}: {count} ({pct:.0f}%)")
    print()

    print("Angle Distribution:")
    for angle, pct in sorted(result['angle_distribution'].items(),
                             key=lambda x: x[1], reverse=True):
        if pct > 0:
            print(f"  {angle}: {pct:.1f}%")
    print()

    print(f"Differentiation Score: {result['differentiation_score']:.2f}")
    print(f"Recommended Angle: {result['recommended_angle']}")
    print()

    if args.verbose and result.get('top_competitors'):
        print("Top Competitors:")
        for i, ch in enumerate(result['top_competitors'][:5], 1):
            print(f"  {i}. {ch['channel_name']} ({ch['video_count']} videos, "
                  f"{ch['total_views']:,} views)")


if __name__ == '__main__':
    main()
```
